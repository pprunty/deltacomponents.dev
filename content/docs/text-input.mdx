## Examples

### Default Variant

```tsx
import { TextInput } from "@/components/ui/text-input"

export default function Example() {
  return (
    <TextInput
      label="Username"
      name="username"
      placeholder="johndoe"
      required
    />
  )
}
```

### Pill Variant

```tsx
import { TextInput } from "@/components/ui/text-input"

export default function Example() {
  return (
    <TextInput
      label="Email"
      name="email"
      type="email"
      variant="pill"
      coloredBorder={true}
      placeholder="john@example.com"
    />
  )
}
```

### With Validation

```tsx
import { z } from "zod"
import { useState } from "react"
import { TextInput } from "@/components/ui/text-input"

export default function ValidationExample() {
  const [error, setError] = useState<string | undefined>()
  
  // Schema to validate email format
  const schema = z.string().email({
    message: "Please enter a valid email address",
  })
  
  const handleValidate = (isValid: boolean, value: string, error?: string) => {
    setError(error)
  }
  
  return (
    <TextInput
      label="Email"
      name="email"
      type="email"
      schema={schema}
      onValidate={handleValidate}
      error={error}
      hint="We'll never share your email with anyone else"
      required
    />
  )
}
```

## Accessibility

The TextInput component is built with accessibility in mind:

- Proper labeling and ARIA attributes
- Support for required field indicators
- Error messages are properly associated with the input
- Keyboard navigation support
- Clear focus states for keyboard users
