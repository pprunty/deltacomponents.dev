## Examples

### Default Variant

```tsx
<TextareaInput
  label="Feedback"
  name="feedback"
  description="Share your thoughts with us"
  placeholder="Type your feedback here..."
/>
```

### Pill Variant with Colored Border

```tsx
<TextareaInput
  label="Feedback"
  name="feedback"
  variant="pill"
  coloredBorder={true}
  placeholder="Type your feedback here..."
/>
```

### Different Sizes

```tsx
<TextareaInput
  label="Small Textarea"
  name="small-textarea"
  size="sm"
  placeholder="Small size textarea"
/>

<TextareaInput
  label="Medium Textarea"
  name="medium-textarea"
  size="md"
  placeholder="Medium size textarea"
/>

<TextareaInput
  label="Large Textarea"
  name="large-textarea"
  size="lg"
  placeholder="Large size textarea"
/>
```

## Validation

The TextareaInput component supports Zod validation:

```tsx
import { z } from "zod";
import { useState } from "react";
import { TextareaInput } from "@/components/ui/textarea-input";

export default function TextareaValidationExample() {
  const [error, setError] = useState<string | undefined>();
  
  // Schema to validate the text has a minimum length
  const schema = z.string().min(10, {
    message: "Feedback must be at least 10 characters",
  });
  
  const handleValidate = (isValid: boolean, value: string, error?: string) => {
    setError(error);
  };
  
  return (
    <TextareaInput
      label="Feedback"
      name="feedback"
      schema={schema}
      onValidate={handleValidate}
      error={error}
      hint="Please provide detailed feedback"
    />
  );
}
```

## Accessibility

The TextareaInput component is built with accessibility in mind:

- Proper labeling and ARIA attributes
- Clear focus states for keyboard users
- Error messages are properly associated with the textarea
- Properly handles disabled states
