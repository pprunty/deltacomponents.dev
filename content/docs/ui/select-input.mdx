# SelectInput

A customizable select input component that supports both controlled and uncontrolled states, with built-in validation support using Zod.

## Examples

### Default Variant

```tsx
import { SelectInput } from "@/components/ui/select-input"

export default function Example() {
  const options = [
    { value: "option1", label: "Option 1" },
    { value: "option2", label: "Option 2" },
    { value: "option3", label: "Option 3" },
  ]

  return (
    <SelectInput
      label="Default Select"
      name="default-select"
      options={options}
      placeholder="Select an option"
    />
  )
}
```

### Pill Variant

```tsx
import { SelectInput } from "@/components/ui/select-input"

export default function Example() {
  const options = [
    { value: "option1", label: "Option 1" },
    { value: "option2", label: "Option 2" },
    { value: "option3", label: "Option 3" },
  ]

  return (
    <SelectInput
      label="Pill Select"
      name="pill-select"
      options={options}
      variant="pill"
      coloredBorder={true}
      placeholder="Select an option"
    />
  )
}
```

### With Validation

```tsx
import { z } from "zod"
import { useState } from "react"
import { SelectInput } from "@/components/ui/select-input"

export default function ValidationExample() {
  const [error, setError] = useState<string | undefined>()
  
  const options = [
    { value: "option1", label: "Option 1" },
    { value: "option2", label: "Option 2" },
    { value: "option3", label: "Option 3" },
  ]
  
  // Schema to validate required selection
  const schema = z.string().min(1, {
    message: "Please select an option",
  })
  
  const handleValidate = (isValid: boolean, value: string, error?: string) => {
    setError(error)
  }
  
  return (
    <SelectInput
      label="Validated Select"
      name="validated-select"
      options={options}
      schema={schema}
      onValidate={handleValidate}
      error={error}
      required
      hint="A selection is required"
    />
  )
}
```

## Accessibility

The SelectInput component is built with accessibility in mind:

- Proper labeling and ARIA attributes
- Support for required field indicators
- Error messages are properly associated with the select
- Keyboard navigation support
- Clear focus states for keyboard users
- Proper contrast ratios for all states
