## Usage

```tsx
"use client"

import { SwitchInput } from "@/registry/inputs/switch-input"

export function Component() {
  return (
    <SwitchInput
      label="Enable notifications"
      name="notifications"
      onCheckedChange={(checked) => console.log(checked)}
    />
  )
}
```

## API Reference

### Props

| Prop                | Type                                                         | Default         | Description                                 |
| ------------------- | ------------------------------------------------------------ | --------------- | ------------------------------------------- |
| `label`             | `string`                                                     | -               | The label text for the switch               |
| `name`              | `string`                                                     | -               | Form field name for submission              |
| `checked`           | `boolean`                                                    | -               | Controlled checked state                    |
| `defaultChecked`    | `boolean`                                                    | `false`         | Uncontrolled default state                  |
| `onCheckedChange`   | `(checked: boolean) => void`                                 | -               | Callback when state changes                 |
| `disabled`          | `boolean`                                                    | `false`         | Whether the switch is disabled              |
| `required`          | `boolean`                                                    | `false`         | Whether the field is required               |
| `variant`           | `"pill" \| "rectangular"`                                    | `"rectangular"` | Switch shape variant                        |
| `size`              | `"default" \| "large"`                                       | `"default"`     | Switch size (24px / 30px height)            |
| `activeColor`       | `string`                                                     | -               | Custom CSS color for active state           |
| `extendedClickArea` | `boolean`                                                    | `false`         | Extends clickable area for better mobile UX |
| `description`       | `string`                                                     | -               | Optional description text                   |
| `hint`              | `string`                                                     | -               | Optional hint text                          |
| `error`             | `string`                                                     | -               | Error message to display                    |
| `schema`            | `z.ZodType<boolean>`                                         | -               | Zod schema for validation                   |
| `onValidate`        | `(isValid: boolean, value: boolean, error?: string) => void` | -               | Validation callback                         |
