## Usage

```tsx
"use client"

import { TagsInput } from "@/registry/inputs/tags-input"

export function Component() {
  return (
    <TagsInput
      label="Tags"
      name="tags"
      onChange={(tags) => console.log(tags)}
    />
  )
}
```

## API Reference

### Props

| Prop            | Type                                                          | Default     | Description                             |
| --------------- | ------------------------------------------------------------- | ----------- | --------------------------------------- |
| `label`         | `string`                                                      | -           | The label text for the input            |
| `name`          | `string`                                                      | -           | Form field name for submission          |
| `value`         | `string[]`                                                    | -           | Controlled tags array                   |
| `defaultValue`  | `string[]`                                                    | `[]`        | Uncontrolled default tags               |
| `onChange`      | `(tags: string[]) => void`                                    | -           | Callback when tags change               |
| `triggerKey`    | `"Enter" \| "Space" \| "Comma"`                               | `"Enter"`   | Key that triggers tag addition          |
| `variant`       | `"default" \| "pill"`                                         | `"default"` | Input styling variant                   |
| `coloredBorder` | `boolean`                                                     | `false`     | Show colored border (pill variant only) |
| `maxTags`       | `number`                                                      | -           | Maximum number of tags allowed          |
| `disabled`      | `boolean`                                                     | `false`     | Whether the input is disabled           |
| `required`      | `boolean`                                                     | `false`     | Whether the field is required           |
| `description`   | `string`                                                      | -           | Optional description text               |
| `hint`          | `string`                                                      | -           | Optional hint text                      |
| `error`         | `string`                                                      | -           | Error message to display                |
| `schema`        | `z.ZodType<string[]>`                                         | -           | Zod schema for validation               |
| `onValidate`    | `(isValid: boolean, value: string[], error?: string) => void` | -           | Validation callback                     |
