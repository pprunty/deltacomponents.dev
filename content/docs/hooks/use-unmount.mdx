## Usage

```tsx {1, 4-6}
import { useUnmount } from "@/hooks/use-unmount"

export default function Component() {
  useUnmount(() => {
    // Cleanup logic here
  })

  return <div>Hello world</div>
}
```

## API Reference

### Parameters
